class Worker:
    def __init__(self,x,y,q,rate,lam,batch_size):
        self.x = x
        self.y = y
        self.q = q
        self.rate = rate
        self.lam = lam
        self.batch_size = batch_size
        self.w = numpy.zeros(shape = [x.shape[1],1])
    
    def set_param(self,w):
        self.w = w
    
    def loss(self):
        yx = numpy.multiply(self.y,self.x)
        yxw = numpy.dot(yx,self.w)
        e = numpy.exp(-yxw)
        log = numpy.log(1+e)
        return numpy.sum(log)
    
    def gradient(self,indict):
        
        origin_w = self.w
        for i in range(q):
            t_x = self.x[indict*self.batch_size:(indict+1)*self.batch_size,:]
            t_y = self.y[indict*self.batch_size:(indict+1)*self.batch_size,:]
            indict = indict+1
            indict = indict%(math.ceil(self.x.shape[0]/self.batch_size))
            yx = numpy.multiply(t_y,t_x)
            yxw = numpy.dot(yx,self.w)
            e = numpy.exp(yxw)
            div = numpy.divide(-yx,1+e)
            g = numpy.sum(div,axis=0).reshape(t_x.shape[1],1)
            g = g/self.batch_size
            self.w = self.w - self.rate*(g+self.lam*self.w)
        p = origin_w - self.w
        return p, indict

class Server:
    def __init__(self,m,n,d,rate,lam):
        self.lam = lam
        self.m = m
        self.n = n
        self.d = d
        self.rate = rate
        self.w = numpy.zeros(shape = [self.d,1])
        self.g = numpy.zeros_like(self.w)
        self.loss = 0
        self.obj = 0
    
    def aggregate(self, grads, losses):
        self.g = numpy.zeros_like(self.w)
        self.loss = 0
        for k in range(self.m):
            self.g = self.g+grads[k]
            self.loss += losses[k]
    
    def gradient(self):
        self.g = self.g/self.m + self.lam*self.w
        
    def update_weight(self):
        self.w -= self.rate*self.g
    
    def broadcast(self):
        return self.w
    
    def objective(self):
        self.obj = self.loss/self.n
        return self.obj

import math
def create_server_workers(m,x,y,rate,lam,q,batch_size):
    n,d = x.shape
    s = math.floor(n/m)
    server = Server(m,n,d,rate,lam)
    workers =[]
    
    for i in range(m):
        
        indices = list(range(i*s, (i+1)*s))
        worker = Worker(x[indices,:],y[indices,:],q,rate,lam,batch_size)
        workers.append(worker)
    
    return server,workers

def model(server, workers, max_communication, n, batch_size):
    obj = []
    indict = 0
    for j in range(max_communication):
        w = server.broadcast()
        grade = []
        for i in range(m):
            workers[i].set_param(w)
            grade.append(0)
        loss = []
        for i in range(m):
            grade[i], indict=workers[i].gradient(indict)
            loss.append(workers[i].loss())
        server.aggregate(grade,loss)
        server.gradient()
        server.update_weight()
        obj.append(server.objective())
    return obj
