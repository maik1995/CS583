def to_one_hot(y, num_class=10):
    temp = numpy.zeros(shape=(y.shape[0],num_class))
    for i in range(y.shape[0]):

from keras.layers import GlobalMaxPooling2D, ZeroPadding2D, Conv2D, MaxPooling2D, Flatten, Dense, Dropout, BatchNormalization, Activation, Input, Add
from keras.models import Sequential, Model

def resnet_identity_block(x,filters,kernel_size,triger = 0):
    x_skip = x
    x = Conv2D(filters=filters,kernel_size=(kernel_size,kernel_size),padding='same')(x)
    if triger == 1:
        x = BatchNormalization(axis=3)(x)
    x = Activation('relu')(x)
    x = Conv2D(filters=filters,kernel_size=(kernel_size,kernel_size),padding='same')(x)
    if triger == 1:
        x = BatchNormalization(axis=3)(x)
    x = Activation('relu')(x)
    x = Add()([x,x_skip])
    x = Activation('relu')(x)
    return x

def resnet_cov_block(x,filters,kernel_size,triger = 0):
    x_skip = Conv2D(filters=filters,kernel_size=(kernel_size,kernel_size),padding='same')(x)
    if triger == 1:
        x_skip = BatchNormalization(axis=3)(x_skip)
    x_skip = Activation('relu')(x_skip)
    x = Conv2D(filters=filters,kernel_size=(kernel_size,kernel_size),padding='same')(x)
    if triger == 1:
        x = BatchNormalization(axis=3)(x)
    x = Activation('relu')(x)
    x = Add()([x,x_skip])
    x = Activation('relu')(x)
    return x

def auxiliary_output(x):
    x = GlobalMaxPooling2D()(x)
    x = Dropout(0.2)(x)
    x = Dense(200)(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    x = Dropout(0.2)(x)
    x = Dense(200)(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    x = Dense(10, activation='softmax')(x)

def resnet(shape = (32,32,3)):
    x_input = Input(shape)
    x = x_input
    x = Conv2D(32,(3,3),padding='same',input_shape=shape, name="input_layor")(x)
    x = BatchNormalization(axis=3)(x)
    x = Activation('relu')(x)
    x = resnet_identity_block(x,32,3,0)
    x = resnet_identity_block(x,32,3,0)
    x = MaxPooling2D(2,2)(x)
    x = Dropout(0.5)(x)
    x = resnet_cov_block(x,64,3,0)
    x = resnet_identity_block(x,64,3,0)
    x = resnet_identity_block(x,64,3,0)
    aux_output_0 = auxiliary_output(x)
    x = MaxPooling2D(2,2)(x)
    x = Dropout(0.5)(x)
    x = resnet_cov_block(x,128,3,1)
    x = resnet_identity_block(x,128,3,1)
    x = resnet_identity_block(x,128,3,1)
    x = GlobalMaxPooling2D()(x)
    x = Dropout(0.2)(x)
    x = Dense(500)(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    x = Dropout(0.5)(x)
    x = Dense(500)(x)
    x = BatchNormalization()(x)
    x = Activation('relu')(x)
    final = Dense(10, activation='softmax')(x)
    model = Model(inputs = x_input, outputs = (aux_output_0,final))
    return model

def batch_generator(x,y,batch_size):
    step = 0
    while True:
        if step == x.shape[0]:
            step = 0
        x_temp = numpy.zeros(shape=(batch_size,x.shape[1],x.shape[2],x.shape[3]))
        y1_temp = numpy.zeros(shape=(batch_size,y.shape[1]))
        y2_temp = numpy.zeros(shape=(batch_size,y.shape[1]))
        for i in range(batch_size):
            x_temp[i] = x[step]
            y1_temp[i] = y[step]
            y2_temp[i] = y[step]
            step+=1
        yield(x_temp,[y1_temp,y2_temp])
